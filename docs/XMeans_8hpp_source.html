304  double XMeans::bayesianInformationCriterion(const std::vector<DataType> &dataset, const std::vector<std::vector<std::size_t> > &indices_list, const std::vector<std::vector<double> > &centroids)
305  {
306  double bic( std::numeric_limits<double>::max() );
307 
308  /* 計算に使うので先にdoubleにキャストしておく */
309  double dim(m_dim);
310  double K(indices_list.size());
311  double N(0);
312  double squared_sigma(0.0);
313 
314  // calc variance
315  for (std::size_t cluster_id = 0; cluster_id < indices_list.size(); ++cluster_id)
316  {
317  const std::vector<double> centroid(centroids.at(cluster_id));
318  for (std::size_t data_index = 0; data_index < indices_list.at(cluster_id).size(); ++data_index)
319  {
320  DataType data = dataset.at(indices_list.at(cluster_id).at(data_index));
321  squared_sigma += m_kmeans.calcSquaredDistance(data, centroid);
322  }
323 
324  N += static_cast<double>(indices_list.at(cluster_id).size());
325  }
326 
327  if ( N - K > 0 )
328  {
329  squared_sigma /= (N - K);
330  double p = (K - 1) + dim * K + 1;
331 
332  // in case of the same points, sigma_sqrt can be zero
333  double sigma_multiplier(0.0);
334  if (squared_sigma <= 0.0)
335  {
336  sigma_multiplier = -1.0 * std::numeric_limits<double>::max();
337  }
338  else
339  {
340  sigma_multiplier = dim * 0.5 * std::log(squared_sigma);
341  }
342 
343  // splitting criterion
344  bic = 0.0;
345  for (std::size_t cluster_id = 0; cluster_id < indices_list.size(); ++cluster_id)
346  {
347  double n = static_cast<double>(indices_list.at(cluster_id).size());
348  double L = n * std::log(n) - n * std::log(N) - n * 0.5 * std::log(2.0 * M_PI) - n * sigma_multiplier - (n - K) * 0.5;
349  bic += p * 0.5 * std::log(N) - L;
350  }
351  }
352 
353  return bic;
354  }
